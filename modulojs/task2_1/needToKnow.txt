Finding if an element exists in the array or not and updating the arraySection
function pushInArray (array, newElement) {
    if (array.indexOf(newElement) === -1) {
        array.push(newElement);
        console.log('New array collection is : ' + array);
    } else if (array.indexOf(newElement) > -1) {
        console.log(newElement + ' already exists in the array collection.');
    }
}

var somearray = ['potato', 'tomato', 'chillies', 'green-pepper'];

pushInArray(somearray, 'spinach'); 
// New array collection is : potato,tomato,chillies,green-pepper,spinach
pushInArray(somearray, 'spinach'); 
// spinach already exists in the array collection
